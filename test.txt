def session():
    con = sl.connect('data.db')
    with con:
        sql_data = con.execute("select count(*) from sqlite_master where type='table' and name='points'")
        for i in sql_data:
            if i[0] == 0:
                con.execute("""
                    CREATE TABLE points (
                    date TEXT,
                    time TEXT PRIMARY KEY,
                    distance REAL,
                    SH REAL
                    );
                    """)

    return con

def stop_session():
    con = session()
    with con:
        con.execute("DELETE FROM points")
    con.close()
